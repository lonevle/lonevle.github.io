SQL常用命令

2023/1/8

USE

USE database -- 使用数据库

SELECT

SELECT * FROM table -- 查询table表的全部数据
SELECT ID FROM table -- 查询table表id列的数据
SELECT ID,ID * 10 FROM table -- 查询table表id及id计算后的数据
SELECT ID,ID * 10 AS 'newID' FROM table -- 查询table表id及id计算后重命名的数据
SELECT 10 * 10 -- 查询计算结果

WHERE

SELECT * FROM table ID = 1 -- 筛选条件

ORDER

SELECT * FROM table ORDER BY ID -- 按id排序（升序）
SELECT * FROM table ORDER BY ID DESC -- 按id降序

DISTINCT

SELECT DISTINCT name FROM table -- 查询name去重后的结果

AND OR

SELECT * FROM table ID > 1 AND ID < 10 -- 查询id大于1并且小于10
SELECT * FROM table ID > 1 OR year = '2006-01-02' -- 查询i大于1或者年份为2006-01-02

NOT

SELECT * FROM table NOT ID > 10 -- 查询id小于等于10
-- 条件否定 eg： NOT IN / NOT LIKE / NOT BETWEEN

IN

SELECT * FROM table ID IN (10,20,30) -- 查询id为10，20，30
SELECT * FROM table ID NOT IN (10,20,30) -- 查询id不为10，20，30

BETWEEN

SELECT * FROM table ID [NOT] BETWEEN 10 AND 20 -- 查询id[不]在10和20之间

LIKE

SELECT * FROM table name LIKE 'p%' -- 以p开始的名字
SELECT * FROM table name LIKE '%p' -- 以p结尾
SELECT * FROM table name LIKE '%p%' -- 包含p
SELECT * FROM table name LIKE '__p' -- 第二个字符为p
-- % 匹配任意数量的任意字符
-- _ 匹配单个任意字符

REGEXP 正则

SELECT * FROM table name REGEXP 'pp|nn' -- 使用正则

IS

SELECT * FROM table name IS NULL -- 匹配name为空
SELECT * FROM table name IS NOT NULL -- 匹配name不为空
-- NULL 空
-- NOT NULL 不为空

LIMIT

SELECT * FROM table name IS NOT NULL LIMIT 10 -- name不为空的10条
SELECT * FROM table name IS NOT NULL LIMIT 20, 10 -- 跳过20条，再取10条
-- 用于分页显示

JOIN

SELECT * FROM table JOIN table2 ON table.id = table2.id -- 内联表查询
SELECT
    * 
FROM table a 
LEFT JOIN table1 b
    ON a.id = b.id
-- 外连表查询 左连接
SELECT
    * 
FROM table a 
RIGHT JOIN table1 b
    ON a.id = b.id
-- 外连表查询 右连接
SELECT name, a.id FROM table a JOIN table2 b ON a.id = b.id -- 添加别名
SELECT name, a.id FROM database.table a JOIN database2.table2 b ON a.id = b.id -- 不同数据库表连接查询
SELECT a.name,b.name FROM table a JOIN table b ON table.id = table2.user -- 单表连接
SELECT
    * 
FROM table a 
JOIN table1 b
    ON a.id = b.id
JOIN table2 c
    ON a.name = c.name 
-- 多表连接查询
SELECT
    * 
FROM table a 
JOIN table1 b
    ON a.id = b.id
    AND a.name = b.name
-- 多条件连接

MockingBird

2022/9/19

MockingBird使用

Anaconda安装

到镜像站下载
Index of /anaconda/archive/ | 清华大学开源软件镜像站 | Tsinghua Open Source Mirror
一般下一步到完成

配置Anaconda源

conda config --add channels https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/free/
conda config --add channels https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/main/ 
conda config --set show_channel_urls yes

创建环境(需要python3.8以上环境运行MockingBird)

conda create -n 环境名 python=3.8

如果创建不成功，尝试修改 %USERPROFILE%\.condarc 文件为以下配置

channels:
  - https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/main/win-64/
  - https://mirrors.tuna.tsinghua.edu.cn/anaconda/pkgs/free/win-64/
  - https://mirrors.tuna.tsinghua.edu.cn/anaconda/cloud/pytorch/win-64/

ssl_verify: false # 重点
show_channel_urls: true

查看所有环境

conda info --envs

进入环境

conda activate 环境名

退出环境

deactivate 环境名

删除环境

conda remove -n 环境名 --all

详细安装参考Anaconda使用教程一（新手友好） - 知乎 (zhihu.com)

下载cuda
https://developer.nvidia.com/cuda-downloads

安装cuda

选择自定义安装，可只勾选勾选cuda，不勾选vs code

查看安装版本

nvcc --version

安装 PyTorch 本地|派托奇 (pytorch.org)

成功执行以下命令则进行下一步

import torch
torch.cuda.is_available()

下载 ffmpeg ，解压后将 bin 目录添加到环境变量
Releases · BtbN/FFmpeg-Builds (github.com)
查看版本

ffmpeg -version

下载项目

git clone git@github.com:babysor/MockingBird.git

在项目目录下执行

pip install -r requirements.txt
pip install webrtcvad-wheels

若下载速度慢可以使用以下命令后重新下载

pip install pqi
pqi use aliyun

安装过程报错 Microsoft Visual C++ 14.0 or greater is required
需要安装Microsoft C++ 生成工具 - Visual Studio

下载模型到 MockingBird\synthesizer\saved_models 目录
https://pan.baidu.com/s/1PI-hM3sn5wbeChRryX-RCQ 提取码：2021

执行 python demo_toolbox.py 启动
录制一小段录音，输入文字，即可播放
若加载模型出错
MockingBird\synthesizer\utils\symbols.py 文件修改

_characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz12340!\'(),-.:;? '

MockingBird\synthesizer\hparams.py 文件修改

use_gst = False
use_ser_for_gst = False
